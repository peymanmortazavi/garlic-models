23  template<GARLIC_VIEW Layer>
24  inline bool
25  emit(yaml_emitter_t* emitter,
26  yaml_event_t* event,
27  Layer&& layer,
28  char* buffer,
29  yaml_mapping_style_t mapping_style = yaml_mapping_style_t::YAML_ANY_MAPPING_STYLE,
30  yaml_sequence_style_t sequence_style = yaml_sequence_style_t::YAML_ANY_SEQUENCE_STYLE) {
31  if (layer.is_object()) {
32  if (!internal::emit_mapping_start(emitter, event, mapping_style))
33  return false;
34  for (const auto& pair : layer.get_object()) {
35  if (!emit(emitter, event, pair.key, buffer, mapping_style, sequence_style))
36  return false;
37  if (!emit(emitter, event, pair.value, buffer, mapping_style, sequence_style))
38  return false;
39  }
40  yaml_mapping_end_event_initialize(event);
41  } else if (layer.is_list()) {
42  if (!internal::emit_sequence_start(emitter, event, sequence_style))
43  return false;
44  for (const auto& item : layer.get_list()) {
45  emit(emitter, event, item, buffer, mapping_style, sequence_style);
46  }
47  yaml_sequence_end_event_initialize(event);
48  } else if (layer.is_bool()) {
49  if (! (layer.get_bool() ? internal::initialize_true(event) : internal::initialize_false(event)) )
50  return false;
51  } else if (layer.is_int()) {
52  if (!internal::initialize_int(event, layer.get_int(), buffer))
53  return false;
54  } else if (layer.is_double()) {
55  if (!internal::initialize_double(event, layer.get_double(), buffer))
56  return false;
57  } else if (layer.is_string()) {
58  auto view = layer.get_string_view();
59  if (!internal::initialize_string(event, view.data(), view.size()))
60  return false;
61  } else {
62  if (!internal::initialize_null(event))
63  return false;
64  }
65  if (!yaml_emitter_emit(emitter, event))
66  return false;
67 
68  return true;
69  }

71  template<GARLIC_VIEW Layer, typename Initializer>
72  static inline tl::expected<void, EmitterProblem>
73  emit(Layer&& layer, bool pretty, Initializer&& initializer) {
74  yaml_emitter_t emitter;
75  yaml_event_t event;
76 
77  if (!yaml_emitter_initialize(&emitter))
78  return tl::make_unexpected(EmitterProblem{emitter.problem});
79 
80  if (!internal::emit_start_stream_and_document(&emitter, &event)) {
81  auto problem = EmitterProblem { emitter.problem };
82  yaml_emitter_delete(&emitter);
83  return tl::make_unexpected(problem);
84  }
85 
86  initializer(&emitter);
87 
88  auto mapping_style = yaml_mapping_style_t::YAML_ANY_MAPPING_STYLE;
89  auto sequence_style = yaml_sequence_style_t::YAML_ANY_SEQUENCE_STYLE;
90  if (pretty) {
91  mapping_style = yaml_mapping_style_t::YAML_BLOCK_MAPPING_STYLE;
92  sequence_style = yaml_sequence_style_t::YAML_BLOCK_SEQUENCE_STYLE;
93  }
94 
95  char buffer[325]; // enough to serializer smallest and largest double values.
96  if (emit(&emitter, &event, layer, buffer, mapping_style, sequence_style) &&
97  internal::emit_end_stream_and_document(&emitter, &event)) {
98  yaml_emitter_delete(&emitter);
99  return tl::expected<void, EmitterProblem>();
100  }
101 
102  auto problem = EmitterProblem { emitter.problem };
103  yaml_emitter_delete(&emitter);
104  return tl::make_unexpected(problem);
105  }
